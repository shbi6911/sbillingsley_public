def toMeters(length,units):
    '''This function will convert an input length to meters, provided it has that length in its dictionary
        Parameters
        -----------
        length = the length to be converted
        units = a string value of the SI abbreviation of the units of the "length" parameter
        Returns
        ---------
        meters = the value of "length" when converted to meters
        Example
        --------
        x = toMeters(100,'cm')
        print(x)                 #should print 1.0'''
    if units in list(conversions.keys()):
        print(f"(converting from {units} to meters)")
        meters = length * conversions[units]
        return meters
    else:
        print('You have entered an unknown unit.')
        return 0

import numpy as np
def planck (T, w_units, units='nm'):
    """
    The Planck function, with flexible units.
        Parameters
        ----------
        T : float
        A temperature, in units of K
        w_units : np.array
        A numpy array of wavelengths, in any units
        units : str
        A string stating the units of the wavelengths.
        
        Functions called
        ----------------
        toMeters
        
        Returns
        -------
        flux : np.array
        The Planck flux at the wavelengths.
    """
    h = 6.626e-34 # in J*s
    c = 2.998e8   # in m/s
    k = 1.381e-23 # in J/K
    
    w = toMeters(w_units,units)         #convert to meters
    
    return 2 * h * c**2 / (w**5 * (np.exp(h * c / (w * k * T))- 1))    #calculate PLanck function
